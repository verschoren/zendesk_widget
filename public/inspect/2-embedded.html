Widget embedded directly into a page container

1. Floating 2. Embedded 3. Multi-Column 4. Custom Launcher

Expected Behavior

The widget should render inside the container below with full messaging interface. Uses autorender: false and explicit render with mode: 'embedded'. No floating launcher button should appear.

Widget Target Element

Implementation Code

<!-- Widget container in HTML -->
<div id="widget" style="width: 100%; height: 600px;"></div>

<!-- Configure autorender: false -->
<script>
  window.zEMessenger = { autorender: false }
</script>

<!-- Load Zendesk snippet -->
<script id="ze-snippet" src="https://static.zdassets.com/ekr/snippet.js?key=e125418a-9466-44d8-9b3f-2ac10e911ea4"></script>

<script>
  // Wait for zE to be available
  const checkWidget = setInterval(() => {
    if (window.zE) {
      clearInterval(checkWidget)

      // Render widget in embedded mode
      window.zE('messenger', 'render', {
        mode: 'embedded',
        widget: {
          targetElement: '#widget'
        }
      }, (error) => {
        if (error) {
          console.error('❌ Render error:', error)
        } else {
          console.log('✅ Widget rendered successfully')
        }
      })
    }
  }, 100)

  setTimeout(() => clearInterval(checkWidget), 10000)
</script>

Key Configuration

// Embedded mode configuration
{
  mode: 'embedded',           // Required for embedded rendering
  widget: {
    targetElement: '#widget'  // CSS selector for container
  }
}

// Important: Set autorender to false BEFORE loading snippet
window.zEMessenger = { autorender: false }
