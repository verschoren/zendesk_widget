Implementation Code

<!-- HTML containers -->
<div id="conversation-list" style="height: 600px;"></div>
<div id="message-log" style="height: 600px;"></div>

<!-- Configure autorender: false -->
<script>
  window.zEMessenger = { autorender: false }
</script>

<!-- Load Zendesk snippet -->
<script id="ze-snippet" src="https://static.zdassets.com/ekr/snippet.js?key=e125418a-9466-44d8-9b3f-2ac10e911ea4"></script>

<script>
  // Wait for zE to be available
  const checkWidget = setInterval(() => {
    if (window.zE) {
      clearInterval(checkWidget)

      // Render with separate conversation list and message log
      window.zE('messenger', 'render', {
        mode: 'embedded',
        conversationList: {
          targetElement: '#conversation-list'
        },
        messageLog: {
          targetElement: '#message-log'
        }
      }, (error) => {
        if (error) {
          console.error('❌ Render error:', error)
        } else {
          console.log('✅ Multi-column widget rendered')
        }
      })
    }
  }, 100)

  setTimeout(() => clearInterval(checkWidget), 10000)
</script>

Key Configuration

// Multi-column configuration
{

  conversationList: {
    targetElement: '#conversation-list'  // Left container
  },
  messageLog: {
    targetElement: '#message-log'        // Right container
  }
}
