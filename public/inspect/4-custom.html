4. Custom Launcher with Metadata Loading...

Multi-column embed with chrome disabled and custom new conversation button

1. Floating 2. Embedded 3. Multi-Column 4. Custom Launcher

Expected Behavior

Widget renders without default chrome/headers. Custom "New Conversation" button creates a new conversation with pre-filled metadata (source, topic) and an initial message. This pattern is ideal for deeply integrated custom support experiences.

Conversation List (No Header)

Message Log (No Header)

Implementation Code

<!-- HTML containers -->
<div id="conversation-list"></div>
<div id="message-log"></div>
<button id="newConversation">New Conversation</button>

<!-- Configure autorender: false -->
<script>
  window.zEMessenger = { autorender: false }
</script>

<!-- Load Zendesk snippet -->
<script id="ze-snippet" src="https://static.zdassets.com/ekr/snippet.js?key=YOUR_KEY"></script>

<script>
  const checkWidget = setInterval(() => {
    if (window.zE) {
      clearInterval(checkWidget)

      // Render with chrome disabled
      window.zE('messenger', 'render', {
        mode: 'embedded',
        conversationList: { targetElement: '#conversation-list' },
        messageLog: { targetElement: '#message-log' }
      })

      // Hide default headers
      window.zE('messenger:set', 'customization', {
        common: { hideHeader: false },
        conversationList: { hideHeader: true },
        messageLog: { hideHeader: true }
      })
    }
  }, 100)

  // Custom new conversation button
  document.getElementById('newConversation').addEventListener('click', () => {
    if (window.zE) {
      // Create new conversation with metadata and pre-filled message
      window.zE('messenger:ui', 'newConversation', {
        displayName: 'Support Request',
        metadata: {
          source: 'demo_page',
          topic: 'custom_launcher'
        },
        message: {
          content: {
            type: 'text',
            text: 'How does the embeddable web widget work?'
          }
        }
      })
    }
  })
</script>

Key Configuration

// Hide chrome/headers
window.zE('messenger:set', 'customization', {
  common: { hideHeader: false },           // Keep main header
  conversationList: { hideHeader: true },  // Hide list header
  messageLog: { hideHeader: true }        // Hide message header
})

// Create new conversation with metadata and pre-filled message
window.zE('messenger:ui', 'newConversation', {
  displayName: 'Support Request',        // Conversation title
  metadata: {

    topic: 'custom_launcher'
  },
  message: {
    content: {

    }
  }
